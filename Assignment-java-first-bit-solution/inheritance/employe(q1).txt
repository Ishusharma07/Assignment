package inheritance;

class employee {
    int id;
    String name;
    double salary;
    static int count = 0;

    employee() {
        this.id = 1;
        this.name = "kanu";
        this.salary = 23000;
        count++;
    }

    employee(int n , String g , double s){
        this.id = n;
        this.name = g;
        this.salary = s;
        count++;
    }

    void setid(int i) {
        this.id = i;
    }

    void setname(String s) {
        this.name = s;
    }
    void setsalary(double d) {
        this.salary = d;
    }

    int getid() {
        return this.id;
    }

    String getname() {     // FIXED
        return this.name;
    }

    double getsalary(){
        return this.salary;
    }

    static int getCount() {
        return count;
    }

    void display() {
        System.out.println("id = " + this.id);
        System.out.println("name = " + this.name);
        System.out.println("salary = " + this.salary);
    }
}

class hr extends employee {
    String department;
    int teamsize;

    String getDepartment() {
        return department;
    }
    void setDepartment(String department) {
        this.department = department;
    }
    int getTeamsize() {
        return teamsize;
    }
    void setTeamsize(int teamsize) {
        this.teamsize = teamsize;
    }

    hr(){
        super();
        this.department = "Human Resources";
        this.teamsize = 5;
    }

    hr(int id , String name , double salary , String s , int i){
        super(id , name , salary);
        this.department = s;
        this.teamsize = i;
    }

    void display() {
        super.display();
        System.out.println("department : " + this.department);
        System.out.println("teamsize : " + this.teamsize);
    }
}
